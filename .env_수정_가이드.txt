# =====================================================
# 📝 server/.env 파일 수정 가이드
# =====================================================

현재 데이터베이스 연결 실패 문제를 해결하려면
server/.env 파일을 다음과 같이 수정하세요:

# -----------------------------------------------------
# 🔧 방법 1: DATABASE_URL 사용 (가장 권장!) ⭐
# -----------------------------------------------------

1. Render Dashboard 접속
2. PostgreSQL 데이터베이스 선택
3. "Info" 탭에서 "External Database URL" 복사
4. server/.env 파일을 다음과 같이 수정:

DATABASE_URL=복사한_URL_여기에_붙여넣기
PORT=3000
NODE_ENV=development
FRONTEND_URL=http://localhost:5173

# 기존의 DB_HOST, DB_PORT, DB_NAME, DB_USER, DB_PASSWORD는 삭제하거나 주석 처리


# -----------------------------------------------------
# 🔧 방법 2: 개별 환경 변수 수정
# -----------------------------------------------------

DB_HOST=dpg-d41atm75r7bs739b60o0-a.oregon-postgres.render.com
DB_PORT=5432
DB_NAME=order_app_db_asrw
DB_USER=order_app_db_asrw_user
DB_PASSWORD=실제_비밀번호_입력

NODE_ENV=development

# =====================================================
# 🚀 수정 후 실행할 명령어
# =====================================================

1. cd server
2. npm run init-db
3. npm run dev (새 터미널)
4. node test-functions.js (프로젝트 루트에서)
